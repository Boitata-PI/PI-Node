<template>
  <!-- Conteúdo principal -->
  <main class="view-container">
    <h1>Detalhes da Disciplina</h1>

    <!-- Abas de navegação -->
    <div class="tabs">
      <button 
        class="tab-button" 
        :class="{ active: activeTab === 'detalhes' }" 
        @click="activeTab = 'detalhes'">
        Detalhes
      </button>
      <button 
        class="tab-button" 
        :class="{ active: activeTab === 'grupos' }" 
        @click="activeTab = 'grupos'">
        Grupos
      </button>
      <button 
        class="tab-button" 
        :class="{ active: activeTab === 'alunos' }" 
        @click="activeTab = 'alunos'">
        Alunos
      </button>
    </div>

    <!-- Conteúdo das abas -->
    <div v-if="activeTab === 'detalhes'" class="details-container">
      <div class="detail-item">
        <strong>Código da Disciplina:</strong>
        <span>{{ disciplina.codigo }}</span>
      </div>
      <div class="detail-item">
        <strong>Nome da Disciplina:</strong>
        <span>{{ disciplina.nome }}</span>
      </div>
      <div class="detail-item">
        <strong>Professor Responsável:</strong>
        <span>{{ disciplina.professor }}</span>
      </div>
      <div class="button-container">
        <router-link :to="'/editar-disciplina/' + disciplina.codigo">
          <button class="action-btn edit-btn">Editar</button>
        </router-link>
        <button @click="deleteDisciplina" class="action-btn delete-btn">Excluir</button>
      </div>
    </div>

    <div v-if="activeTab === 'grupos'" class="tab-content">
      <p>Listagem de grupos relacionados à disciplina.</p>
    </div>

    <div v-if="activeTab === 'alunos'" class="tab-content">
      <p>Listagem de alunos matriculados nesta disciplina.</p>
    </div>
  </main>
</template>

<script>
export default {
  data() {
    return {
      disciplina: {
        codigo: '',
        nome: '',
        professor: '',
      },
      activeTab: 'detalhes', // Aba ativa inicial
    };
  },
  mounted() {
    this.disciplina = {
      codigo: '101',
      nome: 'Matemática Avançada',
      professor: 'Carlos Oliveira',
    };
  },
  methods: {
    async deleteDisciplina() {
      try {
        alert(`Disciplina ${this.disciplina.nome} excluída com sucesso!`);
        this.$router.push('/disciplinas');
      } catch (error) {
        console.error('Erro ao excluir disciplina:', error);
        alert('Erro ao tentar excluir a disciplina.');
      }
    },
  },
};
</script>
